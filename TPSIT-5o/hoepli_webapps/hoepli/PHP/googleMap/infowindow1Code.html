<SCRIPT>
   // varibili globali 
var map; // dichiarazione oggetto map 
   var marker; // dichiarazione oggetto marker 
   var infowindow; // dichiarazione variabile infowindow
   var image; // icona per il marker 
   var stringa; // striga HTML da visualzizare 
   
   //funzione richiamata all'apertura del body
   function inizializza(){
   // definizione coordinate di centratura della mappa
   coordinate = new google.maps.LatLng(45.812, 9.0855);
   // dichiarazione opzioni della mappa
   var opzioni = { 
   zoom: 10, // livello di zoom della mappa mostrata
   center: coordinate,
   mapTypeId: google.maps.MapTypeId.SATELLITE // tipo di mappa 
   }
   // creazione dell'oggetto mappa vero e proprio 
   map = new google.maps.Map(document.getElementById('mappa'), opzioni);
   // simbolo dei marker
   image = {
   url: 'mark.png', 
   size: new google.maps.Size(32, 32),
   origin: new google.maps.Point(0, 0),
   anchor: new google.maps.Point(16, 16)
   };
   // definizione del marker
   marker = new google.maps.Marker({
   position: coordinate,
   map: map,
   icon: image
   });
   // testo html da visualizzare 
   stringa='<br><b>Benvenuti a Como</b><br>il lago più bello del mondo!';
   // creazione istanza della InfoWindows
   infowindow = new google.maps.InfoWindow({
   content: stringa, // stringa HTML mostrata nella infowindow
   position: coordinate // anche differenti da quelle del marker 
   });
   // metodo open che mostra la marker con infowindow 
   infowindow.open(map, marker); 
   }
 </SCRIPT>
