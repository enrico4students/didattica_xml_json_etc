var napoli   = new google.maps.LatLng(40.8333,14.25);
var castello = new google.maps.LatLng(40.8277,14.2478);

   url: 'img/omino.png',                            // file dell'immagine 
   size:   new google.maps.Size(32, 32),        // dimensione del marker in pixel
   origin: new google.maps.Point(0, 0),         // origine 
   anchor: new google.maps.Point(16, 16)        // punto marker collocato in LatLng
 };
 var mapOptions = {    
  zoom: 13,    
  center: napoli  
};  
var map;
   var marker;
function inizializza(){ 
     
   map = new google.maps.Map(document.getElementById('map-canvas'), mapOptions); 
   
   marker = new google.maps.Marker({
   map: map, // la mappa appena creata 
   title:"il castello", // stringa con descrizione del marker 
   Snippet:" da visitare!", // testo posizionato sottoil titolo 
   position: castello, // posizioniamo il markher a queste coordinate 
   icon: immagine
   }); 
 addMarker(marker, map); // aggiunge il marker sulla mappa 
}
google.maps.event.addDomListener(window, 'load', inizializza);
