<title>Creazione di un array di oggetti in Javascript con JSON</title>
<script language="javascript">
document.writeln("<h2>Array di oggetti JSON</h2>");
var libri = { 
 "informatica" : [
  { "linguaggio" : "C++", "prezzo" : 15.50},
  { "linguaggio" : "JavaScript", "prezzo" : 17.50},
  { "linguaggio" : "Java", "prezzo" : 12.50},
 ],
 "cucina" : [
  { "portata" : "antipasti", "prezzo" : 25.50},
  { "portata" : "dolci", "prezzo" : 13.00}
 ]
}
var i = 0
document.writeln("<table border='2'><tr>");
for(i = 0; i < libri.informatica.length; i++)
{
 document.writeln("<td>");
 document.writeln("<table border='1' width=100 >");
 document.writeln("<tr><td><b>linguaggio</b></td><td width=100>"+libri.informatica[i].linguaggio+"</td></tr>");
 document.writeln("<tr><td><b>prezzo</b></td><td width=100>"+ libri.informatica[i].prezzo +"</td></tr>");
 document.writeln("</table>");
 document.writeln("</td>");
}
for(i = 0; i < libri.cucina.length; i++)
{
 document.writeln("<td>");
 document.writeln("<table border='1' width=100 >");
 document.writeln("<tr><td><b>portata</b></td><td width=100>"+libri.cucina[i].portata+"</td></tr>");
 document.writeln("<tr><td><b>prezzo</b></td><td width=100>" + libri.cucina[i].prezzo+"</td></tr>");
 document.writeln("</table>");
 document.writeln("</td>");
}
document.writeln("</tr></table>");
</script>
