<h2>Crea un oggetto JSON da una stringa </h2>
<script>
var testo = '{"autori":[' +
'{"nome":"Paolo","cognome":"Camagni" },' +
'{"nome":"Riccardo","cognome":"Nikolassy" }]}';
var oggetto = eval ("(" + testo + ")");
document.write("<h3>Il primo autore = " + oggetto.autori[0].nome +"<\/h3>");
document.write("<h3>Il secondo autore = " + oggetto.autori[1].nome +"<\/h3>");
</script>
